open Crowbar

(* Define the fuzzer test case *)
let test_of_string =
  Crowbar.add_test ~name:"Hs_code.of_string naive parser resilience" [ bytes ] (fun random_string ->
      (* Run the function with random string input *)
      match Hs_code.of_string random_string with
      | Ok _hs_code ->
          (* If it succeeds, awesome! *)
          ()
      | Error _msg ->
          (* If it safely returns a Result.Error, that's a controlled failure *)
          ())
