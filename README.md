CPTPP rules of origin and tariffs calculator.

**This calculator is for intuition/informational/education purposes, not for actual legal work. Please read the BSD 3-Clause License before usage because it contains important legal information.**

The calculator uses the [official CPTPP text](https://www.mfat.govt.nz/en/trade/free-trade-agreements/free-trade-agreements-in-force/cptpp/comprehensive-and-progressive-agreement-for-trans-pacific-partnership-text-and-resources).
## Installation
Download the zip, cd to the folder then:

```bash
dune build
opam install . --deps-only --with-test
dune build
```

## How the calculator works
### Step 0: get tariff data
The calculator requires harmonized tariff data generated by the [cptpp-parser](https://github.com/nguyenhoangan-monkey/cptpp-parser). 

* **If building from source:** You do not need to download this manually. Running `make` or your first build script will automatically fetch the latest pristine dataset from the parser tracking repository into the local `/data` directory.
* **If installing via package manager:** The dataset is natively bundled directly inside the distribution package—no setup required.


### Step 1: create objects and build accumulation tree
In this example below, I defined a (hypothetical) Hatsune Miku figurine that is officially licensed and has a very high profit margin, and then record it as a `hatsune_miku` variable.

There are two ways to create an object. The first way is the unsafe, explicit way. This is recommended for quick back-of-the-napkin math and experimentation where you don't mind programs being crashed.

```ocaml
open Domain

(* HS code -> Country -> material value USD -> Material.t *)
let pvc_pellet = Material.of_strings_exn "3904.10" "China" "5.00"
let paint      = Material.of_strings_exn "3208.20" "Japan" "2.50"
let box        = Material.of_strings_exn "4819.10" "Vietnam" "1.50"

(* HS code -> free on board value USD -> departure country -> arrival country -> Good.t *)
let hatsune_miku = Good.of_strings_exn "9503.00.00" "20.00" "Vietnam" "Mexico"

(* Creating an accumulation tech tree with custom descriptions *)
let unsafe_miku_tree =
  Tech_tree.good "Hatsune Miku Figurine (初音ミク)" hatsune_miku
    [
      Tech_tree.material "PVC Plastic Pellets" pvc_pellet [];
      Tech_tree.material "Acrylic Paint Sub-component" paint [];
      Tech_tree.material "Cardboard Packaging Box" box [];
    ]
```

In practice, you don't want to use `of_string_exn`. You want to use safe `of_string` and perform error handling. The second way is the safe, error handling way. This is recommended for actual production-grade work. In this example, the free-on-board value is adjustable as `fob_str`:

```ocaml
let make_miku destination fob_str =
  let open Result.Syntax in

  let+ pvc_pellet = Material.make Material.{ 
    hs_code = Raw "3904.10"; 
    origin  = Raw "China"; 
    cost    = Raw "5.00" 
  }
  and+ paint = Material.make Material.{ 
    hs_code = Raw "3208.20"; 
    origin  = Raw "Japan"; 
    cost    = Raw "2.50" 
  }
  and+ box = Material.make Material.{ 
    hs_code = Raw "4819.10"; 
    origin  = Raw "Vietnam"; 
    cost    = Raw "1.50" 
  }
  and+ hatsune_miku = Good.make Good.{
    hs_code             = Raw "9503.00.00";
    free_on_board_value = Raw fob_str; 
    shipped_from        = Raw "VNM";
    shipped_to          = Ready destination; 
  } in

  (* Accumulation tech tree using localized and international trade identifiers *)
  Tech_tree.good "Hatsune Miku Figurine (初音ミク) - Juguete de plástico 🇯🇵" hatsune_miku
    [
      Tech_tree.material "PVC Plastic Pellets / 聚氯乙烯树脂颗粒" pvc_pellet [];
      Tech_tree.material "Acrylic Paint Sub-component / アクリル塗料" paint [];
      Tech_tree.material "Cardboard Packaging Box / Thùng carton đóng gói" box [];
    ]

(* Pre-validating destination to build a specialized factory pipeline *)
let make_miku_mexico = make_miku (Country.of_string_exn "Mexico")
```

Then we can manipulate this function. We have `Raw fob_str` to check input, so if functions don't work, `spec_validation` returns `None`:

```ocaml
(** Generate products at different price points **)
let spec_validation = function
  | Ok tree -> Some tree
  | Error err -> 
      Printf.eprintf "[ERROR] Specification failed: %s\n" err;
      None

let raw_price_points = ["20.00"; "30.00"; "100.00"]

let miku_tree_list =
  raw_price_points
  |> List.map make_miku_mexico
  |> List.filter_map spec_validation
```

Another example demonstrating how data is processed, as `Raw` (raw user input), `Maybe` (a lazy stream, from an API or a generator) and `Ready` (evaluated proper types). It is worth noting that most software assumes that:
* goods can be made of materials since it is an assembly
* goods can be made of other goods, since it is a higher order assembly

However, in this engine:
* material can made of goods (fully localized, origin certified)
* material can be made of other materials (same origin or unknown origin raw material)

```ocaml
(* Note: Akita Neru is a derivative from Hatsune Miku *)
(* We take Hatsune Miku box, put it to another box with Akita Neru graphics *)

(* Under non-qualifying operations, this is obviously a non-compliant CPTPP product *)
(* Aka we are committing origin fraud *)
let generate_akita_neru_tree api_data_thunk underlying_miku_tree =
  let open Result.Syntax in

  (* 1. Material got from MercadoLibre, got from Developer API via lazy thunks *)
  let api_data = api_data_thunk () in
  let* box_material = Material.make Material.{ 
    hs_code = Maybe api_data.api_hs; 
    origin  = Maybe api_data.api_country; 
    cost    = Maybe api_data.api_price;
  } in

  (* 2. Material added by user, user don't know exactly the source *)
  let* paper_material = Material.make Material.{ 
    hs_code = Raw "48";
    origin  = Raw "Guatemala"; 
    cost    = Raw "0.10" 
  } in

  (* 3. Test multilingual processing and Ready type *)
  let japan = Country.of_string_exn "日本" in
  let mexico = Country.of_string_exn "メキシコ" in
  
  let* akita_neru_good = Good.make Good.{
    hs_code             = Raw "9503.00.00";
    free_on_board_value = Raw "50.00";
    shipped_from        = Ready mexico;
    shipped_to          = Ready japan;
  } in

  (* Assemble the fully resolved structural tree matrix *)
  Tech_tree.good "Akita Neru para Mercadolibre" akita_neru_good
    [
      underlying_miku_tree; (* Explicitly nested sub-assembly branch *)
      Tech_tree.material "Caja de Mercadolibre MX 🇲🇽/ Cardboard box" box_material
        [Tech_tree.material "Papel (paper)" paper_material []];
    ]
```
